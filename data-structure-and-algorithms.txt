/*Encuentra el mayor (max) Dado un array (var array PARAMETRO) con múltiples valores, escribe una función (function) que devuelva(return) el número mayor (ej: para [-3,3,5,7] el número mayor (max) es 7).  */

function max(array){
  var currentMax = array[0];
 
  for(var i = 0; i <= array.length-1; i++){
    if(array[i] > currentMax){
      currentMax = array[i];
    }
 
  }
 
  return currentMax;
}
 
var array = [-3,3,5,7];
 
console.log(max(array));

Max/Min/Avg - Dado un array con múltiples valores, escribe una función que devuelva un nuevo array que solo contenga el valor mayor (max), menor (min) y promedio (avg) del array original (ej: [1,5,10,-2] devolverá [10,-2,3.5]).

function maxMinAvg(arr){
  var suma = 0;
  var newArray = [];
  var currentMax = arr[0];
  var currentMin = arr[0];
 
  /* PROMEDIO */
 
  for(var i = 0; i < arr.length; i++){
    suma = suma + arr[i];
   
  }
 
  var promedio = suma/arr.length;
 
  /* MAX */
 
  for(var i=0; i<arr.length; i++){
    if(arr[i] > currentMax){
      currentMax = arr[i];
    }
  }
 
  /* MIN */
 
  for(var i=0; i<arr.length; i++){
    if(arr[i] < currentMin){
      currentMin = arr[i];
    }
  }
 
 
  newArray.push(currentMax);
  newArray.push(currentMin);
  newArray.push(promedio);
 
  return newArray;
 
}
 
var array = [1,5,10,-2];
 
console.log(maxMinAvg(array));

*/
